def _pick_col(df, candidates):
    for c in candidates:
        if c in df.columns:
            return c
    df_cols_norm = {col.strip().lower(): col for col in df.columns}
    for c in candidates:
        key = c.strip().lower()
        if key in df_cols_norm:
            return df_cols_norm[key]
    return None

def _normalize_columns(df: pd.DataFrame) -> pd.DataFrame:
    if df is None:
        return df
    d = df.copy()
    d.columns = [(" ".join(str(c).replace("\ufffd", "").split())).strip() for c in d.columns]
    if d.columns.duplicated().any():
        d = d.loc[:, ~d.columns.duplicated()].copy()
    return d

def _ensure_dir(path):
    folder = os.path.dirname(path)
    if folder and not os.path.exists(folder):
        os.makedirs(folder, exist_ok=True)

def _normalize_status_bin(df, state_col):
    if state_col and state_col in df.columns:
        s = df[state_col].astype(str).fillna("").str.strip().str.lower()
        closed_mask = (
            s.str.contains("closed", na=False) | s.str.contains("fech", na=False) |
            s.str.contains("resolv", na=False) | s.str.contains("complete", na=False) |
            s.str.contains("done", na=False) | s.str.contains("encerr", na=False)
        )
        df["Status_Bin"] = closed_mask.map(lambda x: "Fechado" if x else "Aberto")
        return df
    if "Closed" not in df.columns:
        df["Closed"] = pd.NaT
    df["Status_Bin"] = df["Closed"].apply(lambda x: "Fechado" if pd.notna(x) else "Aberto")
    return df

def _ensure_identity_cols_common(df):
    number_col = _pick_col(df, ["Number", "Número", "Ticket", "ID", "Task"])
    if "Number" not in df.columns:
        df["Number"] = df[number_col].astype(str).fillna("") if number_col else ""

    requester_col = _pick_col(df, ["Requester", "Requested for", "Solicitante", "Caller"])
    df["Requester"] = df[requester_col].astype(str).fillna("—") if requester_col else "—"

    ag_col = _pick_col(df, ["Assignment group", "Assignment Group", "Grupo", "Grupo de Atribuição"])
    df["Assignment group"] = df[ag_col].astype(str).fillna("") if ag_col else ""

    asg_col = _pick_col(df, ["Assigned to", "AssignedTo", "Analista", "Assignee"])
    df["Assigned to"] = df[asg_col].astype(str).fillna("") if asg_col else ""

    if "State" not in df.columns:
        state_col = _pick_col(df, ["Status", "Estado"])
        df["State"] = df[state_col] if state_col else None

    return df

def load_prepare_reqs(source_file, label="REQs", now=None, q=None):
    log_q(q, f"\n📂 [{label}] Carregando dados de: {source_file}")
    if not os.path.exists(source_file):
        raise FileNotFoundError(f"❌ Arquivo não encontrado: {source_file}")

    df = pd.read_excel(source_file)
    if df.columns.duplicated().any():
        df = df.loc[:, ~df.columns.duplicated()]
    log_q(q, f"   ✓ {len(df)} registros carregados")

    opened_col = _pick_col(df, ["Opened", "Open", "Data Abertura", "Created"])
    closed_col = _pick_col(df, ["Resolved", "Resolved at", "ResolvedAt", "Closed", "Close", "Data Fechamento"])
    state_col = _pick_col(df, ["State", "Status", "Estado", "Status Bin", "Status_Bin"])

    df["Opened"] = pd.to_datetime(df[opened_col], errors="coerce") if opened_col else pd.NaT
    df["Closed"] = pd.to_datetime(df[closed_col], errors="coerce") if closed_col else pd.NaT
    df = _normalize_status_bin(df, state_col)

    now = now or datetime.now()
    df["Aging_Timedelta"] = df.apply(lambda x: (x["Closed"] - x["Opened"]) if pd.notna(x["Closed"]) else (now - x["Opened"]), axis=1)
    df["Aging"] = (df["Aging_Timedelta"].dt.total_seconds() / (24 * 3600)).fillna(0).astype(int)

    return _ensure_identity_cols_common(df)

def build_month_window_html(dfs, months_limit=None):
    opened_months, closed_months = [], []
    for df in dfs:
        if df is None or df.empty:
            continue
        mo = pd.to_datetime(df["Opened"], errors="coerce").dropna()
        mc = pd.to_datetime(df["Closed"], errors="coerce").dropna()
        if not mo.empty:
            opened_months.extend([mo.min(), mo.max()])
        if not mc.empty:
            closed_months.extend([mc.min(), mc.max()])

    valid_dates = [d for d in (opened_months + closed_months) if pd.notna(d)]
    if not valid_dates:
        now_dt = datetime.now()
        start = pd.Timestamp(year=now_dt.year, month=now_dt.month, day=1)
        return pd.DatetimeIndex([]), {"periodo_inicio": "N/A", "periodo_fim": "N/A", "janela": "N/A", "meses_calculado": 0, "start_date": start, "end_date": pd.Timestamp(now_dt)}

    month_start = min(valid_dates).replace(day=1, hour=0, minute=0, second=0, microsecond=0)
    month_end = max(valid_dates).replace(day=1, hour=0, minute=0, second=0, microsecond=0)

    all_months = pd.date_range(month_start, month_end, freq="MS")

    if months_limit is not None:
        try:
            months_limit = int(months_limit)
            if months_limit > 0 and len(all_months) > months_limit:
                all_months = all_months[-months_limit:]
        except:
            pass

    if len(all_months) == 0:
        now_dt = datetime.now()
        start = pd.Timestamp(year=now_dt.year, month=now_dt.month, day=1)
        return pd.DatetimeIndex([]), {"periodo_inicio": "N/A", "periodo_fim": "N/A", "janela": "N/A", "meses_calculado": 0, "start_date": start, "end_date": pd.Timestamp(now_dt)}

    final_start = pd.Timestamp(year=all_months[0].year, month=all_months[0].month, day=1)
    final_end = pd.Timestamp(year=all_months[-1].year, month=all_months[-1].month, day=calendar.monthrange(all_months[-1].year, all_months[-1].month)[1])

    meta = {
        "periodo_inicio": all_months[0].strftime("%Y-%m"),
        "periodo_fim": all_months[-1].strftime("%Y-%m"),
        "janela": f"{all_months[0].strftime('%m/%Y')} → {all_months[-1].strftime('%m/%Y')}" if len(all_months) > 1 else all_months[0].strftime('%m/%Y'),
        "meses_calculado": int(len(all_months)),
        "start_date": final_start,
        "end_date": final_end
    }
    return all_months, meta

def compute_open_backlog_metrics_html(df, sla_days, ticket_type="REQ"):
    now = pd.Timestamp(datetime.now())

    if ticket_type == "REQ":
        open_df = _get_only_open_reqs(df)
    else:
        open_df = _get_only_open_incs(df)

    open_df["Opened_dt"] = pd.to_datetime(open_df["Opened"], errors="coerce")
    open_df = open_df[open_df["Opened_dt"].notna()].copy()
    open_df["AGING_OPEN"] = (now - open_df["Opened_dt"]).dt.days
    open_df["AGING_OPEN"] = pd.to_numeric(open_df["AGING_OPEN"], errors="coerce").fillna(0).astype(int).clip(lower=0)
    open_df["SLA_BREACH_OPEN"] = open_df["AGING_OPEN"] > int(sla_days)

    total_open = int(len(open_df))
    breach_open = int(open_df["SLA_BREACH_OPEN"].sum()) if total_open else 0

    if ticket_type == "REQ":
        bins = [-1, 3, 5, 7, 10, 20, 30, np.inf]
        labels = ["1-3 dias", "3-5 dias", "5-7 dias", "7-10 dias", "10 dias +", "20 dias +", "30 dias +"]
    else:
        bins = [-1, 3, 5, 7, np.inf]
        labels = ["1-3 dias", "3-5 dias", "5-7 dias", "7+ dias"]

    if total_open > 0:
        open_df["AgingBin"] = pd.cut(open_df["AGING_OPEN"], bins=bins, labels=labels, include_lowest=False, right=True)
        bucket_counts = open_df.groupby("AgingBin").size().reindex(labels, fill_value=0)
    else:
        bucket_counts = pd.Series(index=labels, data=np.zeros(len(labels), dtype=int))

    bucket_total = int(bucket_counts.sum())
    bucket_pct = ((bucket_counts / bucket_total * 100) if bucket_total > 0 else 0).round(1)
    buckets_df = pd.DataFrame({"Faixa": labels, "Qtd": bucket_counts.values.astype(int), "%": bucket_pct.values})

    kpis = {
        "open_total": total_open,
        "open_breach": breach_open,
        "mttr_mean": round(float(open_df["AGING_OPEN"].mean()) if total_open else 0.0, 1),
    }
    return kpis, buckets_df, None, open_df

def compute_closed_metrics_html(df, ticket_type="REQ"):
    d = df.copy()
    if ticket_type == "REQ":
        state_col = _pick_col(d, ["State", "Status", "Estado"])
        if state_col:
            s = d[state_col].astype(str).str.strip().str.lower()
            closed_df = d[(s.isin(["closed complete", "closed incomplete"]) | s.str.contains("closed complet|closed incomplet", na=False)) & (d["Opened"].notna())].copy()
        else:
            closed_df = d[(d["Closed"].notna()) & (d["Opened"].notna())].copy()
    else:
        state_col = _pick_col(d, ["State", "Status", "Estado"])
        if state_col:
            s = d[state_col].astype(str).str.strip().str.lower()
            closed_df = d[(s.isin(["resolved", "canceled", "cancelled", "closed"]) | s.str.contains("resolv|cancel|clos", na=False)) & (d["Opened"].notna())].copy()
        else:
            closed_df = d[(d["Closed"].notna()) & (d["Opened"].notna())].copy()

    if len(closed_df) == 0:
        return {"closed_total": 0, "mttr_mean": 0.0}

    closed_dt_safe = pd.to_datetime(closed_df["Closed"]).fillna(pd.Timestamp(datetime.now()))
    closed_df["AGING_CLOSED"] = (closed_dt_safe - pd.to_datetime(closed_df["Opened"])).dt.days
    closed_df["AGING_CLOSED"] = pd.to_numeric(closed_df["AGING_CLOSED"], errors="coerce").fillna(0).astype(int).clip(lower=0)

    return {
        "closed_total": len(closed_df),
        "mttr_mean": round(float(closed_df["AGING_CLOSED"].mean()), 1),
    }

def get_current_month_daily(df_daily):
    if df_daily.empty:
        return df_daily
    d = df_daily.copy()
    d["_dt"] = pd.to_datetime(d["Data"], format="%d/%m/%Y")
    max_month = d["_dt"].dt.to_period("M").max()
    filtered = d[d["_dt"].dt.to_period("M") == max_month].drop(columns=["_dt"]).copy()
    return filtered

def compute_daily_table_html(df, start_date=None, end_date=None, ticket_type="REQ", real_backlog=0, sla_days=0):
    if df is None or df.empty:
        return pd.DataFrame(columns=["Data", "Abertos", "Resolvidos", "Cancelados", "NetFlow", "Backlog Acumulado", "SLA Perdido"])

    d = df.copy().reset_index(drop=True)
    d["State_Lower"] = d["State"].astype(str).str.lower().str.strip() if "State" in d.columns else ""
    opened_dates = pd.to_datetime(d["Opened"], errors="coerce").dropna().dt.floor("D")

    d_closed = d[d["Closed"].notna()].copy()
    d_closed["Closed_Date"] = pd.to_datetime(d_closed["Closed"], errors="coerce").dt.floor("D")

    if ticket_type == "INC":
        mask_canceled = d_closed["State_Lower"].isin(["canceled", "cancelled"]) | d_closed["State_Lower"].str.contains(r"\bcancel", regex=True)
        mask_resolved = d_closed["State_Lower"].isin(["resolved", "closed"]) | d_closed["State_Lower"].str.contains("resolv|clos", regex=True)
        mask_resolved = mask_resolved & ~mask_canceled
    else:
        mask_canceled = d_closed["State_Lower"].isin(["closed incomplete"]) | d_closed["State_Lower"].str.contains("closed incomplet", regex=False)
        mask_resolved = d_closed["State_Lower"].isin(["closed complete"]) | d_closed["State_Lower"].str.contains("closed complet", regex=False)

    resolved_dates = d_closed[mask_resolved]["Closed_Date"]
    canceled_dates = d_closed[mask_canceled]["Closed_Date"]
    
    # Lógica de SLA Perdido Diário (apenas informacional)
    mask_sla_breach = mask_resolved & (d_closed["Aging"] > sla_days)
    sla_breach_dates = d_closed[mask_sla_breach]["Closed_Date"]

    valid_dts = [dt for dt in [opened_dates.min(), resolved_dates.min(), canceled_dates.min()] if pd.notna(dt)]
    if not valid_dts and not start_date:
        return pd.DataFrame(columns=["Data", "Abertos", "Resolvidos", "Cancelados", "NetFlow", "Backlog Acumulado", "SLA Perdido"])

    min_dt = min(valid_dts) if valid_dts else start_date
    max_dt = end_date if end_date else datetime.now()

    all_days_global = pd.date_range(min_dt, max_dt, freq="D")
    op = opened_dates.value_counts().reindex(all_days_global, fill_value=0)
    res = resolved_dates.value_counts().reindex(all_days_global, fill_value=0)
    canc = canceled_dates.value_counts().reindex(all_days_global, fill_value=0)
    sla_b = sla_breach_dates.value_counts().reindex(all_days_global, fill_value=0)

    net = op - (res + canc)
    
    historico_net = net.values
    backlog_retro = np.zeros(len(historico_net))
    current_bklg = real_backlog
    for i in range(len(historico_net) - 1, -1, -1):
        backlog_retro[i] = current_bklg
        current_bklg = current_bklg - historico_net[i]

    df_out = pd.DataFrame({
        "Data": all_days_global,
        "Abertos": op.values.astype(int),
        "Resolvidos": res.values.astype(int),
        "Cancelados": canc.values.astype(int),
        "NetFlow": net.values.astype(int),
        "Backlog Acumulado": backlog_retro.astype(int),
        "SLA Perdido": sla_b.values.astype(int)
    })

    if start_date and end_date:
        full_month_range = pd.date_range(start_date, end_date, freq="D")
        df_month = pd.DataFrame({"Data": full_month_range})
        df_month = pd.merge(df_month, df_out, on="Data", how="left")
        df_month["Abertos"] = df_month["Abertos"].fillna(0)
        df_month["Resolvidos"] = df_month["Resolvidos"].fillna(0)
        df_month["Cancelados"] = df_month["Cancelados"].fillna(0)
        df_month["NetFlow"] = df_month["NetFlow"].fillna(0)
        df_month["SLA Perdido"] = df_month["SLA Perdido"].fillna(0)
        b_series = df_month["Backlog Acumulado"].ffill().bfill()
        df_month["Backlog Acumulado"] = b_series.fillna(0)
        df_out = df_month.copy()

    df_out["Data"] = df_out["Data"].dt.strftime("%d/%m/%Y")
    for c in ["Abertos", "Resolvidos", "Cancelados", "NetFlow", "Backlog Acumulado", "SLA Perdido"]:
        df_out[c] = df_out[c].astype(int)

    return df_out

    resolved_months = d_closed[mask_resolved]["Closed_Month"].value_counts()
    canceled_months = d_closed[mask_canceled]["Closed_Month"].value_counts()
    
    # SLA Perdido Mensal (Informacional)
    mask_sla_breach = mask_resolved & (d_closed["Aging"] > sla_days)
    sla_breach_months = d_closed[mask_sla_breach]["Closed_Month"].value_counts()

    all_months_index = opened_months.index.union(resolved_months.index).union(canceled_months.index).sort_values()

    if len(all_months_index) == 0:
        return pd.DataFrame(columns=["Mês", "Abertos", "Resolvidos", "Cancelados", "NetFlow", "Backlog Acumulado", "SLA Perdido"])

    if months_limit and len(all_months_index) > months_limit:
        all_months_index = all_months_index[-months_limit:]

    df_out = pd.DataFrame(index=all_months_index)
    df_out["Abertos"] = opened_months.reindex(all_months_index, fill_value=0).values.astype(int)
    df_out["Resolvidos"] = resolved_months.reindex(all_months_index, fill_value=0).values.astype(int)
    df_out["Cancelados"] = canceled_months.reindex(all_months_index, fill_value=0).values.astype(int)
    df_out["SLA Perdido"] = sla_breach_months.reindex(all_months_index, fill_value=0).values.astype(int)
    
    df_out["NetFlow"] = df_out["Abertos"] - (df_out["Resolvidos"] + df_out["Cancelados"])
    
    historico_net = df_out["NetFlow"].values
    backlog_retro = np.zeros(len(historico_net))
    
    current_bklg = real_backlog
    for i in range(len(historico_net) - 1, -1, -1):
        backlog_retro[i] = current_bklg
        current_bklg = current_bklg - historico_net[i]

    df_out["Backlog Acumulado"] = backlog_retro.astype(int)
    df_out.reset_index(inplace=True)
    df_out.rename(columns={"index": "Mês"}, inplace=True)
    df_out["Mês"] = df_out["Mês"].dt.strftime("%b/%Y").str.upper()

    return df_out

def get_short_qc_url(cfg, width, height, q=None):
    url = "https://quickchart.io/chart/create"
    payload = {"chart": cfg, "width": width, "height": height, "format": "png"}
    try:
        data = json.dumps(payload).encode('utf-8')
        req = urllib.request.Request(url, data=data, headers={'Content-Type': 'application/json'})
        with urllib.request.urlopen(req) as response:
            res = json.loads(response.read().decode('utf-8'))
            if res.get("success"):
                return res.get("url")
    except Exception as e:
        log_q(q, f"⚠️ Aviso: Falha ao gerar URL curta do gráfico. Erro: {e}")

    return f"https://quickchart.io/chart?w={width}&h={height}&c={urllib.parse.quote(json.dumps(cfg))}"

def add_bar_formatter(cfg, data_array, is_stacked=False):
    """Garante porcentagem e anula valores iguais a zero para ocultar os rótulos de ponta."""
    try:
        if is_stacked and isinstance(data_array, list) and len(data_array) > 0 and isinstance(data_array[0], list):
            totals_per_bar = [sum(x) for x in zip(*data_array)]
            total = sum(totals_per_bar)
            bar_values = totals_per_bar
        else:
            numeric_data = [float(x) if str(x).replace('.','',1).isdigit() else 0 for x in data_array]
            total = sum(numeric_data)
            bar_values = numeric_data
    except Exception:
        bar_values = [0 for _ in data_array] if not is_stacked else []
        total = 0

    formatted_strings = []
    for val in bar_values:
        if total > 0 and val > 0:
            pct = (val / total) * 100
            pct_str = f"{int(pct)}%" if pct.is_integer() else f"{pct:.1f}%"
            formatted_strings.append(f"{int(val)} ({pct_str})")
        else:
            formatted_strings.append("") 

    if "data" in cfg and "labels" in cfg["data"]:
        orig_labels = cfg["data"]["labels"]
        new_labels = []
        for i, lbl in enumerate(orig_labels):
            fs = formatted_strings[i] if i < len(formatted_strings) else ""
            new_labels.append(f"{lbl} | {fs}" if fs else lbl)
        cfg["data"]["labels"] = new_labels

    if "data" in cfg and "datasets" in cfg["data"]:
        for ds in cfg["data"]["datasets"]:
            if "data" in ds:
                ds["data"] = [v if (v is not None and float(v) > 0) else None for v in ds["data"]]

    if "options" not in cfg:
        cfg["options"] = {}
    if "plugins" not in cfg["options"]:
        cfg["options"]["plugins"] = {}
    if "datalabels" not in cfg["options"]["plugins"]:
        cfg["options"]["plugins"]["datalabels"] = {}

    cfg["options"]["plugins"]["datalabels"]["display"] = True

    if "scales" in cfg["options"] and "xAxes" in cfg["options"]["scales"] and len(cfg["options"]["scales"]["xAxes"]) > 0:
        if "ticks" in cfg["options"]["scales"]["xAxes"][0]:
            max_val = max(bar_values) if bar_values else 10
            cfg["options"]["scales"]["xAxes"][0]["ticks"]["suggestedMax"] = max_val * 1.35

    return cfg

                    else:
                        bg_cell = "transparent"
                except:
                    bg_cell = "transparent"
            elif "SLA X%" in df.columns and j == list(df.columns).index("SLA X%"):
                color = ModernColors.DANGER
                font_weight = "bold"
                bg_cell = "transparent"
            else:

            align = "left" if j == 0 else "center"
            html += f"<td style='padding: 6px; border: 1px solid #ddd; text-align: {align}; color: {color}; font-weight: {font_weight}; background-color: {bg_cell};'>{v_str}</td>"

                    {"id": "y2", "position": "right", "gridLines": {"drawOnChartArea": False}, "ticks": {"beginAtZero": True, "precision": 0, "suggestedMax": max_b * 1.2}}
                ]
            }
        }
    }
    total_in = sum(a) if a and a[0] != 'Sem Dados' else 0
    total_out = sum(f) + sum(c) if f and c and f[0] != 'Sem Dados' else 0
    return add_title_to_cfg(cfg, f"ACOMPANHAMENTO DIÁRIO (MÊS VIGENTE) - TOTAL ENTRADAS: {total_in} | SAÍDAS: {total_out}")

# ORQUESTRADOR DE CRIAÇÃO DO RELATÓRIO 2 (COM DADOS REAIS DO EXCEL)
# ============================================================================
def main_simples(configs, q=None):
    log_q(q, "\n" + "=" * 80)
    log_q(q, "🚀 GERANDO RELATÓRIO 2 (NOVO LAYOUT) COM DADOS REAIS")
    log_q(q, "=" * 80)

    try:
        _ensure_dir(OUTPUT_FILE)
        _ensure_dir(OUTPUT_HTML)
        
        SLA_DAYS_REQS = int(configs.get("sla_days_reqs", 10))
        SLA_DAYS_INCS = int(configs.get("sla_days_incs", 7))
        ETA_DATE = configs.get("eta_date", "")
        MONTHS_LIMIT_HTML = int(configs.get("months_limit_html", 2))
        MONTHS_LIMIT_THROUGHPUT = int(configs.get("months_limit_throughput", 6))
        
        RUN_AT = datetime.now()

        # 1. Carrega os Dados Reais
        df_reqs = load_prepare_reqs(SOURCE_FILE, "REQs", now=RUN_AT, q=q)
        df_incs = load_prepare_incs(SOURCE_INCS, "INCs", now=RUN_AT, q=q)
        
        # 2. Normaliza os tickets para o HTML
        req = normalize_ticket_df_html(df_reqs, "REQ")
        inc = normalize_ticket_df_html(df_incs, "INC")

        # 3. Segregação: INC, REQ (Suporte) e UAM (Acessos)
        req["IS_ACCESS"] = req["Assignment group"].apply(_is_access_ticket_html)
        req_support = req[~req["IS_ACCESS"]].copy()
        req_uam = req[req["IS_ACCESS"]].copy()

        inc_open_strict = _get_only_open_incs(inc)
        req_sup_open_strict = _get_only_open_reqs(req_support)
        uam_open_strict = _get_only_open_reqs(req_uam)

        inc_count = len(inc_open_strict)
        req_support_count = len(req_sup_open_strict)
        uam_count = len(uam_open_strict)
        total_backlog_real = inc_count + req_support_count + uam_count

        all_months, meta = build_month_window_html([req, inc], MONTHS_LIMIT_HTML)
        start_date = meta.get("start_date")
        end_date = meta.get("end_date")
        now_floor = pd.Timestamp(datetime.now()).floor('D')
        if end_date and end_date > now_floor:
            end_date = now_floor

        # ========================================================================
        # FUNÇÃO INTERNA PARA PROCESSAR UMA SESSÃO
        # ========================================================================
        def process_section(df_full, df_open, ticket_type, sla_days, real_backlog, is_inc=False):
            df_m = compute_monthly_throughput_table_html(df_full, MONTHS_LIMIT_THROUGHPUT, ticket_type=ticket_type, real_backlog=real_backlog, sla_days=sla_days)
            
            df_d_full = compute_daily_table_html(df_full, start_date=start_date, end_date=end_date, ticket_type=ticket_type, real_backlog=real_backlog, sla_days=sla_days)
            df_d = get_current_month_daily(df_d_full)

            mttr_dict = compute_closed_metrics_html(df_full, ticket_type=ticket_type)
            mttr_val = mttr_dict["mttr_mean"]

            _, df_ag, _, open_df = compute_open_backlog_metrics_html(df_full, sla_days, ticket_type=ticket_type)

            analyst_counts = df_open["Assigned to"].replace("nan", "Nao Assinado").replace("", "N/A").value_counts().head(10)
            df_a = pd.DataFrame({"Analista": analyst_counts.index, "Volume": analyst_counts.values})
            if df_a.empty:
                df_a = pd.DataFrame(columns=["Analista", "Volume"])

            if "SLA_BREACH_OPEN" in open_df.columns:
                sla_counts = open_df["SLA_BREACH_OPEN"].value_counts()
                in_sla = int(sla_counts.get(False, 0))
                out_sla = int(sla_counts.get(True, 0))
            else:
                in_sla, out_sla = 0, 0

            if "SLA_BREACH_OPEN" in open_df.columns:
                df_critical = open_df[open_df["SLA_BREACH_OPEN"] == True].copy()
            else:
                df_critical = pd.DataFrame()

            if not df_critical.empty:
                df_critical = df_critical.sort_values(by="AGING_OPEN", ascending=False).head(50)
                df_critical["Opened_Str"] = df_critical["Opened"].dt.strftime('%d/%m/%Y').fillna("-")
                df_critical["ETA"] = ETA_DATE
                
                # CÁLCULO DA NOVA COLUNA SLA X%
                df_critical["SLA X%"] = ((df_critical["AGING_OPEN"] / sla_days) * 100).astype(int).astype(str) + "%"
                
                for col in ["Number", "Assigned to", "Requester", "State"]:
                    if col not in df_critical.columns:
                        df_critical[col] = "N/A"
                        
                df_critical = df_critical[["Number", "Assigned to", "Requester", "State", "Opened_Str", "AGING_OPEN", "SLA X%", "ETA"]]
                df_critical.columns = ["Ticket", "Assigned to", "Requester", "State", "Opened", "Aging", "SLA X%", "ETA"]
            else:
                df_critical = pd.DataFrame(columns=["Ticket", "Assigned to", "Requester", "State", "Opened", "Aging", "SLA X%", "ETA"])

            return df_m, df_d, df_a, df_ag, df_critical, mttr_val, in_sla, out_sla

        log_q(q, "Processando dados reais de INCs...")
        inc_m, inc_d, inc_a, inc_ag, inc_sla_df, inc_mttr, inc_in_sla, inc_out_sla = process_section(inc, inc_open_strict, "INC", SLA_DAYS_INCS, inc_count, is_inc=True)

        log_q(q, "Processando dados reais de REQs...")
        req_m, req_d, req_a, req_ag, req_sla_df, req_mttr, req_in_sla, req_out_sla = process_section(req_support, req_sup_open_strict, "REQ", SLA_DAYS_REQS, req_support_count, is_inc=False)

        log_q(q, "Processando dados reais de UAMs...")
        uam_m, uam_d, uam_a, uam_ag, uam_sla_df, uam_mttr, uam_in_sla, uam_out_sla = process_section(req_uam, uam_open_strict, "REQ", SLA_DAYS_REQS, uam_count, is_inc=False)

            cfg_diario = build_daily_trend_cfg(df_d, color_main, ModernColors.SUCCESS, ModernColors.GRAY, label_canc)

            )}
            
            {build_grid2(
                build_card("5. MATURIDADE (AGING)", f'<img src="{qc_url_html(cfg_aging, q)}" width="100%">'),

        log_q(q, traceback.format_exc())

# ============================================================================
# ENVIO DE EMAIL INTERATIVO VIA OUTLOOK
# ============================================================================
def enviar_email_relatorio(html_content, caminho_anexo="", q=None):
    log_q(q, "\n--- Preparando envio de E-mail via Outlook ---")

    safe_destinatario = str(destinatario) if destinatario else ""
    safe_copia = str(copia) if copia else ""
    safe_titulo = str(titulo) if titulo else "Relatório Novo Layout de tickets - Data: "
    
    try:
        safe_importance = int(importance) if importance else 2
    except (ValueError, TypeError):
        safe_importance = 2

    try:
        outlook = win32com.client.Dispatch("outlook.application")
        mail = outlook.CreateItem(0)
        
        if safe_destinatario:
            mail.To = safe_destinatario
        else:
            log_q(q, "⚠️ Aviso: Variável 'destinatario' está vazia. O envio pode falhar.")
            
        mail.Importance = safe_importance
        mail.Subject = safe_titulo + f"{datetime.now().strftime('%d/%m/%Y')}"
        mail.HTMLBody = html_content

        if safe_copia:
            mail.CC = safe_copia

        if caminho_anexo and os.path.exists(caminho_anexo):
            absolute_path = os.path.abspath(caminho_anexo)
            mail.Attachments.Add(absolute_path)

        mail.Send()
        
        if caminho_anexo and os.path.exists(caminho_anexo):
            log_q(q, f"✅ E-mail enviado com sucesso (com anexo: {caminho_anexo})!")
        else:
            log_q(q, "✅ E-mail enviado com sucesso (sem anexo)!")

    except Exception as e:
        log_q(q, f"❌ Erro crítico ao enviar e-mail. Erro: {e}")
