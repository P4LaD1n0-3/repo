def _pick_col(df, candidates):
    for c in candidates:
        if c in df.columns:
            return c
    df_cols_norm = {col.strip().lower(): col for col in df.columns}
    for c in candidates:
        key = c.strip().lower()
        if key in df_cols_norm:
            return df_cols_norm[key]
    return None


def _normalize_columns(df: pd.DataFrame) -> pd.DataFrame:
    if df is None:
        return df
    d = df.copy()
    d.columns = [(" ".join(str(c).replace("\ufffd", "").split())).strip()
                 for c in d.columns]
    if d.columns.duplicated().any():
        d = d.loc[:, ~d.columns.duplicated()].copy()
    return d


def _ensure_dir(path):
    folder = os.path.dirname(path)
    if folder and not os.path.exists(folder):
        os.makedirs(folder, exist_ok=True)


def _normalize_status_bin(df, state_col):
    if state_col and state_col in df.columns:
        s = df[state_col].astype(str).fillna("").str.strip().str.lower()
        closed_mask = (
            s.str.contains("closed", na=False) | s.str.contains("fech", na=False) |
            s.str.contains("resolv", na=False) | s.str.contains("complete", na=False) |
            s.str.contains("done", na=False) | s.str.contains(
                "encerr", na=False)
        )
        df["Status_Bin"] = closed_mask.map(
            lambda x: "Fechado" if x else "Aberto")
        return df
    if "Closed" not in df.columns:
        df["Closed"] = pd.NaT
    df["Status_Bin"] = df["Closed"].apply(
        lambda x: "Fechado" if pd.notna(x) else "Aberto")
    return df


def _ensure_identity_cols_common(df):
    number_col = _pick_col(df, ["Number", "Número", "Ticket", "ID", "Task"])
    if "Number" not in df.columns:
        df["Number"] = df[number_col].astype(
            str).fillna("") if number_col else ""

    requester_col = _pick_col(
        df, ["Requester", "Requested for", "Solicitante", "Caller"])
    df["Requester"] = df[requester_col].astype(
        str).fillna("—") if requester_col else "—"

    ag_col = _pick_col(
        df, ["Assignment group", "Assignment Group", "Grupo", "Grupo de Atribuição"])
    df["Assignment group"] = df[ag_col].astype(
        str).fillna("") if ag_col else ""

    asg_col = _pick_col(
        df, ["Assigned to", "AssignedTo", "Analista", "Assignee"])
    df["Assigned to"] = df[asg_col].astype(str).fillna("") if asg_col else ""

    wn_col = _pick_col(df, ["Work notes", "Notas de trabalho"])
    if "Work notes" not in df.columns:
        df["Work notes"] = df[wn_col] if wn_col else None

    ac_col = _pick_col(df, ["Additional comments", "Comentários adicionais"])
    if "Additional comments" not in df.columns:
        df["Additional comments"] = df[ac_col] if ac_col else None

    if "State" not in df.columns:
        state_col = _pick_col(df, ["Status", "Estado"])
        df["State"] = df[state_col] if state_col else None

    return df


def load_prepare_reqs(source_file, label="REQs", now=None, q=None):
    log_q(q, f"\n📂 [{label}] Carregando dados de: {source_file}")
    if not os.path.exists(source_file):
        raise FileNotFoundError(f"❌ Arquivo não encontrado: {source_file}")

    df = pd.read_excel(source_file)
    if df.columns.duplicated().any():
        df = df.loc[:, ~df.columns.duplicated()]
    log_q(q, f"   ✓ {len(df)} registros carregados")

    opened_col = _pick_col(df, ["Opened", "Open", "Data Abertura", "Created"])
    closed_col = _pick_col(
        df, ["Resolved", "Resolved at", "ResolvedAt", "Closed", "Close", "Data Fechamento"])
    state_col = _pick_col(
        df, ["State", "Status", "Estado", "Status Bin", "Status_Bin"])

    df["Opened"] = pd.to_datetime(
        df[opened_col], errors="coerce") if opened_col else pd.NaT
    df["Closed"] = pd.to_datetime(
        df[closed_col], errors="coerce") if closed_col else pd.NaT
    df = _normalize_status_bin(df, state_col)

    now = now or datetime.now()
    df["Aging_Timedelta"] = df.apply(lambda x: (
        x["Closed"] - x["Opened"]) if pd.notna(x["Closed"]) else (now - x["Opened"]), axis=1)
    df["Aging"] = (df["Aging_Timedelta"].dt.total_seconds() /
                   (24 * 3600)).fillna(0).astype(int)

    required_cols = {"Priority": "Moderada", "Category": "Geral",
                     "Subcategory": "Geral", "Severity": "3 - Low"}
    for col, default_val in required_cols.items():
        if col not in df.columns:
            df[col] = default_val

    return _ensure_identity_cols_common(df)

def load_prepare_ptask(source_file, sla_days, now=None, q=None):
    log_q(q, f"\n📂 [PTASK/PRB] Carregando dados de: {source_file}")
    if not os.path.exists(source_file):
        log_q(
            q, f"   ⚠️ Arquivo de PTASK não encontrado: {source_file}. Gerando base vazia.")
        return pd.DataFrame()

    df = pd.read_excel(source_file)
    df = _normalize_columns(df)
    now_dt = now or datetime.now()
    log_q(q, f"   ✓ {len(df)} registros carregados (PTASK/PRB)")

    opened_col = _pick_col(df, ["Opened", "Open", "Data Abertura", "Created"])
    closed_col = _pick_col(
        df, ["Resolved", "Resolved at", "Closed", "Close", "Data Fechamento"])
    state_col = _pick_col(df, ["State", "Status", "Estado"])
    assigned_col = _pick_col(
        df, ["Assigned to", "AssignedTo", "Analista", "Assignee"])
    rca_status_col = _pick_col(df, ["RCA Status", "Status RCA"])

    df["Opened"] = pd.to_datetime(
        df[opened_col], errors="coerce") if opened_col else pd.NaT
    df["Closed"] = pd.to_datetime(
        df[closed_col], errors="coerce") if closed_col else pd.NaT
    df["Assigned to"] = df[assigned_col].astype(str).fillna("—").replace(
        "", "—").replace("nan", "—") if assigned_col else "—"
    df["RCA Status"] = df[rca_status_col].astype(str).fillna(
        "N/A").replace("nan", "N/A") if rca_status_col else "N/A"

    if state_col:
        s = df[state_col].astype(str).fillna("").str.strip().str.lower()
        closed_mask = s.isin(["closed", "complete", "completed", "resolvido",
                              "fechado"]) | s.str.contains("clos|complet|resolv", regex=True)
        df["State_Bin"] = np.where(closed_mask, "Complete", "Open")
    else:
        df["State_Bin"] = np.where(df["Closed"].notna(), "Complete", "Open")

    opened_dt_safe = df["Opened"]
    closed_dt_safe = df["Closed"].fillna(now_dt)

    aging_td = closed_dt_safe - opened_dt_safe
    df["Aging_Days"] = aging_td.dt.total_seconds() / (24*3600)
    df["Aging_Days"] = df["Aging_Days"].fillna(0).astype(int)

    df["SLA_Status"] = np.where(
        df["Aging_Days"] <= sla_days, "Inside SLA", "Outside SLA")

    return df

def build_month_window_html(dfs, months_limit=None):
    opened_months, closed_months = [], []
    for df in dfs:
        if df is None or df.empty:
            continue
        mo = pd.to_datetime(df["Opened"], errors="coerce").dropna()
        mc = pd.to_datetime(df["Closed"], errors="coerce").dropna()
        if not mo.empty:
            opened_months.extend([mo.min(), mo.max()])
        if not mc.empty:
            closed_months.extend([mc.min(), mc.max()])

    valid_dates = [d for d in (opened_months + closed_months) if pd.notna(d)]
    if not valid_dates:
        now_dt = datetime.now()
        start = pd.Timestamp(year=now_dt.year, month=now_dt.month, day=1)
        return pd.DatetimeIndex([]), {"periodo_inicio": "N/A", "periodo_fim": "N/A", "janela": "N/A", "meses_calculado": 0, "start_date": start, "end_date": pd.Timestamp(now_dt)}

    month_start = min(valid_dates).replace(
        day=1, hour=0, minute=0, second=0, microsecond=0)
    month_end = max(valid_dates).replace(
        day=1, hour=0, minute=0, second=0, microsecond=0)

    all_months = pd.date_range(month_start, month_end, freq="MS")

    if months_limit is not None:
        try:
            months_limit = int(months_limit)
            if months_limit > 0 and len(all_months) > months_limit:
                all_months = all_months[-months_limit:]
        except:
            pass

    if len(all_months) == 0:
        now_dt = datetime.now()
        start = pd.Timestamp(year=now_dt.year, month=now_dt.month, day=1)
        return pd.DatetimeIndex([]), {"periodo_inicio": "N/A", "periodo_fim": "N/A", "janela": "N/A", "meses_calculado": 0, "start_date": start, "end_date": pd.Timestamp(now_dt)}

    final_start = pd.Timestamp(
        year=all_months[0].year, month=all_months[0].month, day=1)
    final_end = pd.Timestamp(year=all_months[-1].year, month=all_months[-1].month,
                             day=calendar.monthrange(all_months[-1].year, all_months[-1].month)[1])

    meta = {
        "periodo_inicio": all_months[0].strftime("%Y-%m"),
        "periodo_fim": all_months[-1].strftime("%Y-%m"),
        "janela": f"{all_months[0].strftime('%m/%Y')} → {all_months[-1].strftime('%m/%Y')}" if len(all_months) > 1 else all_months[0].strftime('%m/%Y'),
        "meses_calculado": int(len(all_months)),
        "start_date": final_start,
        "end_date": final_end
    }
    return all_months, meta


def compute_open_backlog_metrics_html(df, sla_days, ticket_type="REQ"):
    now = pd.Timestamp(datetime.now())

    if ticket_type == "REQ":
        open_df = _get_only_open_reqs(df)
    else:
        open_df = _get_only_open_incs(df)

    open_df["Opened_dt"] = pd.to_datetime(open_df["Opened"], errors="coerce")
    open_df = open_df[open_df["Opened_dt"].notna()].copy()
    open_df["AGING_OPEN"] = (now - open_df["Opened_dt"]).dt.days
    open_df["AGING_OPEN"] = pd.to_numeric(
        open_df["AGING_OPEN"], errors="coerce").fillna(0).astype(int).clip(lower=0)
    open_df["SLA_BREACH_OPEN"] = open_df["AGING_OPEN"] > int(sla_days)

    total_open = int(len(open_df))
    breach_open = int(open_df["SLA_BREACH_OPEN"].sum()) if total_open else 0

    if ticket_type == "REQ":
        bins = [-1, 3, 5, 7, 10, 20, 30, np.inf]
        labels = ["1-3 dias", "3-5 dias", "5-7 dias",
                  "7-10 dias", "10 dias +", "20 dias +", "30 dias +"]
    else:
        bins = [-1, 3, 5, 7, np.inf]
        labels = ["1-3 dias", "3-5 dias", "5-7 dias", "7+ dias"]

    if total_open > 0:
        open_df["AgingBin"] = pd.cut(
            open_df["AGING_OPEN"], bins=bins, labels=labels, include_lowest=False, right=True)
        bucket_counts = open_df.groupby(
            "AgingBin").size().reindex(labels, fill_value=0)
    else:
        bucket_counts = pd.Series(
            index=labels, data=np.zeros(len(labels), dtype=int))

    bucket_total = int(bucket_counts.sum())
    bucket_pct = ((bucket_counts / bucket_total * 100)
                  if bucket_total > 0 else 0).round(1)
    buckets_df = pd.DataFrame(
        {"Faixa": labels, "Qtd": bucket_counts.values.astype(int), "%": bucket_pct.values})

    kpis = {
        "open_total": total_open,
        "open_breach": breach_open,
        "mttr_mean": round(float(open_df["AGING_OPEN"].mean()) if total_open else 0.0, 1),
    }
    return kpis, buckets_df, None, open_df


def compute_closed_metrics_html(df, ticket_type="REQ"):
    d = df.copy()
    if ticket_type == "REQ":
        state_col = _pick_col(d, ["State", "Status", "Estado"])
        if state_col:
            s = d[state_col].astype(str).str.strip().str.lower()
            closed_df = d[(s.isin(["closed complete", "closed incomplete"]) | s.str.contains(
                "closed complet|closed incomplet", na=False)) & (d["Opened"].notna())].copy()
        else:
            closed_df = d[(d["Closed"].notna()) & (d["Opened"].notna())].copy()
    else:
        state_col = _pick_col(d, ["State", "Status", "Estado"])
        if state_col:
            s = d[state_col].astype(str).str.strip().str.lower()
            closed_df = d[(s.isin(["resolved", "canceled", "cancelled", "closed"]) | s.str.contains(
                "resolv|cancel|clos", na=False)) & (d["Opened"].notna())].copy()
        else:
            closed_df = d[(d["Closed"].notna()) & (d["Opened"].notna())].copy()

    if len(closed_df) == 0:
        return {"closed_total": 0, "mttr_mean": 0.0}

    closed_dt_safe = pd.to_datetime(closed_df["Closed"]).fillna(
        pd.Timestamp(datetime.now()))
    closed_df["AGING_CLOSED"] = (
        closed_dt_safe - pd.to_datetime(closed_df["Opened"])).dt.days
    closed_df["AGING_CLOSED"] = pd.to_numeric(
        closed_df["AGING_CLOSED"], errors="coerce").fillna(0).astype(int).clip(lower=0)

    return {
        "closed_total": len(closed_df),
        "mttr_mean": round(float(closed_df["AGING_CLOSED"].mean()), 1),
    }


def get_current_month_daily(df_daily):
    if df_daily.empty:
        return df_daily
    d = df_daily.copy()
    d["_dt"] = pd.to_datetime(d["Data"], format="%d/%m/%Y")
    max_month = d["_dt"].dt.to_period("M").max()
    filtered = d[d["_dt"].dt.to_period("M") == max_month].drop(
        columns=["_dt"]).copy()
    return filtered


def compute_daily_table_html(df, start_date=None, end_date=None, ticket_type="REQ", real_backlog=0):
    if df is None or df.empty:
        return pd.DataFrame(columns=["Data", "Abertos", "Resolvidos", "Cancelados", "NetFlow", "Backlog Acumulado"])

    d = df.copy().reset_index(drop=True)
    d["State_Lower"] = d["State"].astype(str).str.lower().str.strip() if "State" in d.columns else ""
    opened_dates = pd.to_datetime(d["Opened"], errors="coerce").dropna().dt.floor("D")

    d_closed = d[d["Closed"].notna()].copy()
    d_closed["Closed_Date"] = pd.to_datetime(d_closed["Closed"], errors="coerce").dt.floor("D")

    if ticket_type == "INC":
        mask_canceled = d_closed["State_Lower"].isin(["canceled", "cancelled"]) | d_closed["State_Lower"].str.contains(r"\bcancel", regex=True)
        mask_resolved = d_closed["State_Lower"].isin(["resolved", "closed"]) | d_closed["State_Lower"].str.contains("resolv|clos", regex=True)
        mask_resolved = mask_resolved & ~mask_canceled
    else:
        mask_canceled = d_closed["State_Lower"].isin(["closed incomplete"]) | d_closed["State_Lower"].str.contains("closed incomplet", regex=False)
        mask_resolved = d_closed["State_Lower"].isin(["closed complete"]) | d_closed["State_Lower"].str.contains("closed complet", regex=False)

    resolved_dates = d_closed[mask_resolved]["Closed_Date"]
    canceled_dates = d_closed[mask_canceled]["Closed_Date"]

    if opened_dates.empty and resolved_dates.empty and canceled_dates.empty:
        return pd.DataFrame(columns=["Data", "Abertos", "Resolvidos", "Cancelados", "NetFlow", "Backlog Acumulado"])

    valid_dts = [dt for dt in [opened_dates.min(), resolved_dates.min(), canceled_dates.min()] if pd.notna(dt)]
    min_dt = min(valid_dts) if valid_dts else start_date
    max_dt = end_date if end_date else datetime.now()

    all_days_global = pd.date_range(min_dt, max_dt, freq="D")
    op = opened_dates.value_counts().reindex(all_days_global, fill_value=0)
    res = resolved_dates.value_counts().reindex(all_days_global, fill_value=0)
    canc = canceled_dates.value_counts().reindex(all_days_global, fill_value=0)

    net = op - (res + canc)
    
    # REVERSE BALANCE CALCULATION (Viagem no Tempo)
    historico_net = net.values
    backlog_retro = np.zeros(len(historico_net))
    
    current_bklg = real_backlog
    for i in range(len(historico_net) - 1, -1, -1):
        backlog_retro[i] = current_bklg
        current_bklg = current_bklg - historico_net[i]

    df_out = pd.DataFrame({
        "Data": all_days_global,
        "Abertos": op.values.astype(int),
        "Resolvidos": res.values.astype(int),
        "Cancelados": canc.values.astype(int),
        "NetFlow": net.values.astype(int),
        "Backlog Acumulado": backlog_retro.astype(int)
    })

    if start_date and end_date:
        full_month_range = pd.date_range(start_date, end_date, freq="D")
        df_month = pd.DataFrame({"Data": full_month_range})
        df_month = pd.merge(df_month, df_out, on="Data", how="left")
        df_month["Abertos"] = df_month["Abertos"].fillna(0)
        df_month["Resolvidos"] = df_month["Resolvidos"].fillna(0)
        df_month["Cancelados"] = df_month["Cancelados"].fillna(0)
        df_month["NetFlow"] = df_month["NetFlow"].fillna(0)
        
        b_series = df_month["Backlog Acumulado"]
        b_series = b_series.ffill() 
        b_series = b_series.bfill()
        df_month["Backlog Acumulado"] = b_series.fillna(0)
        
        df_out = df_month.copy()

    df_out["Data"] = df_out["Data"].dt.strftime("%d/%m/%Y")
    for c in ["Abertos", "Resolvidos", "Cancelados", "NetFlow", "Backlog Acumulado"]:
        df_out[c] = df_out[c].astype(int)

    return df_out


def compute_monthly_throughput_table_html(df, months_limit=None, ticket_type="REQ", real_backlog=0):
    if df is None or df.empty:
        return pd.DataFrame(columns=["Mês", "Abertos", "Resolvidos", "Cancelados", "NetFlow", "Backlog Acumulado"])

    d = df.copy()
    d["Opened"] = pd.to_datetime(d["Opened"], errors="coerce")
    d["Closed"] = pd.to_datetime(d["Closed"], errors="coerce")
    d["State_Lower"] = d["State"].astype(str).str.lower().str.strip() if "State" in d.columns else ""

    opened_months = d["Opened"].dropna().dt.to_period('M').value_counts()

    d_closed = d[d[ "Closed"].notna()].copy()
    d_closed["Closed_Month"] = d_closed["Closed"].dt.to_period('M')

    if ticket_type == "INC":
        mask_canceled = d_closed["State_Lower"].isin(["canceled", "cancelled"]) | d_closed["State_Lower"].str.contains(r"\bcancel", regex=True)
        mask_resolved = d_closed["State_Lower"].isin(["resolved", "closed"]) | d_closed["State_Lower"].str.contains("resolv|clos", regex=True)
        mask_resolved = mask_resolved & ~mask_canceled
    else:
        mask_canceled = d_closed["State_Lower"].isin(["closed incomplete"]) | d_closed["State_Lower"].str.contains("closed incomplet", regex=False)
        mask_resolved = d_closed["State_Lower"].isin(["closed complete"]) | d_closed["State_Lower"].str.contains("closed complet", regex=False)

    resolved_months = d_closed[mask_resolved]["Closed_Month"].value_counts()
    canceled_months = d_closed[mask_canceled]["Closed_Month"].value_counts()

    all_months_index = opened_months.index.union(resolved_months.index).union(canceled_months.index).sort_values()

    if len(all_months_index) == 0:
        return pd.DataFrame(columns=["Mês", "Abertos", "Resolvidos", "Cancelados", "NetFlow", "Backlog Acumulado"])

    if months_limit and len(all_months_index) > months_limit:
        all_months_index = all_months_index[-months_limit:]

    df_out = pd.DataFrame(index=all_months_index)
    df_out["Abertos"] = opened_months.reindex(all_months_index, fill_value=0).values.astype(int)
    df_out["Resolvidos"] = resolved_months.reindex(all_months_index, fill_value=0).values.astype(int)
    df_out["Cancelados"] = canceled_months.reindex(all_months_index, fill_value=0).values.astype(int)
    
    df_out["NetFlow"] = df_out["Abertos"] - (df_out["Resolvidos"] + df_out["Cancelados"])
    
    historico_net = df_out["NetFlow"].values
    backlog_retro = np.zeros(len(historico_net))
    
    current_bklg = real_backlog
    for i in range(len(historico_net) - 1, -1, -1):
        backlog_retro[i] = current_bklg
        current_bklg = current_bklg - historico_net[i]

    df_out["Backlog Acumulado"] = backlog_retro.astype(int)

    df_out.reset_index(inplace=True)
    df_out.rename(columns={"index": "Mês"}, inplace=True)
    df_out["Mês"] = df_out["Mês"].dt.strftime("%b/%Y").str.upper()

    return df_out

def get_short_qc_url(cfg, width, height, q=None):
    """
    Envia a configuração via POST para a API do QuickChart para gerar uma URL CURTA.
    Isso impede que o Outlook quebre a imagem ao responder o e-mail.
    """
    url = "https://quickchart.io/chart/create"
    payload = {
        "chart": cfg,
        "width": width,
        "height": height,
        "format": "png"
    }
    try:
        data = json.dumps(payload).encode('utf-8')
        req = urllib.request.Request(url, data=data, headers={
                                     'Content-Type': 'application/json'})
        with urllib.request.urlopen(req) as response:
            res = json.loads(response.read().decode('utf-8'))
            if res.get("success"):
                return res.get("url")
    except Exception as e:
        log_q(
            q, f"⚠️ Aviso: Falha ao gerar URL curta do gráfico (Fallback ativado). Erro: {e}")

    # Fallback
    return f"https://quickchart.io/chart?w={width}&h={height}&c={urllib.parse.quote(json.dumps(cfg))}"

def add_bar_formatter(cfg, data_array, is_stacked=False):
    """
    Garante a exibição do Número + Porcentagem nos gráficos de barra horizontais.
    Se o gráfico for empilhado (stacked), o cálculo da % usa o total da barra somada.
    """
    try:
        # Se for stacked (como no gráfico misto), precisamos somar os arrays para achar o total geral real
        if is_stacked and isinstance(data_array, list) and len(data_array) > 0 and isinstance(data_array[0], list):
            totals_per_bar = [sum(x) for x in zip(*data_array)]
            total = sum(totals_per_bar)
        else:
            numeric_data = [float(x) if str(x).replace('.','',1).isdigit() else 0 for x in data_array]
            total = sum(numeric_data)
    except Exception:
        numeric_data = [0 for _ in data_array] if not is_stacked else []
        total = 0

    if not is_stacked:
        formatted_strings = []
        for val in numeric_data:
            if total > 0 and val > 0:
                pct = (val / total) * 100
                pct_str = f"{int(pct)}%" if pct.is_integer() else f"{pct:.1f}%"
                formatted_strings.append(f"{int(val)} ({pct_str})")
            elif val == 0:
                formatted_strings.append("")
            else:
                formatted_strings.append(str(int(val)))

        if "data" in cfg and "labels" in cfg["data"]:
            orig_labels = cfg["data"]["labels"]
            new_labels = []
            for i, lbl in enumerate(orig_labels):
                fs = formatted_strings[i] if i < len(formatted_strings) else ""
                new_labels.append(f"{lbl} | {fs}" if fs else lbl)
            cfg["data"]["labels"] = new_labels

    formatter_js = (
        "function(value, context) { "
        "  if (value === 0 || value === null) return ''; "
        f"  var total = {total}; "
        "  if (total === 0) return value; "
        "  var pct = (value / total) * 100; "
        "  var pctStr = (pct % 1 === 0 ? pct : pct.toFixed(1)) + '%'; "
        "  return value + ' (' + pctStr + ')'; "
        "}"
    )

    if "options" not in cfg:
        cfg["options"] = {}
    if "plugins" not in cfg["options"]:
        cfg["options"]["plugins"] = {}
    if "datalabels" not in cfg["options"]["plugins"]:
        cfg["options"]["plugins"]["datalabels"] = {}

    cfg["options"]["plugins"]["datalabels"]["formatter"] = formatter_js
    cfg["options"]["plugins"]["datalabels"]["display"] = True

    if "scales" in cfg["options"] and "xAxes" in cfg["options"]["scales"] and len(cfg["options"]["scales"]["xAxes"]) > 0:
        if "ticks" in cfg["options"]["scales"]["xAxes"][0]:
            if not is_stacked:
                max_val = max(numeric_data) if numeric_data else 10
            else:
                max_val = max(totals_per_bar) if totals_per_bar else 10
            cfg["options"]["scales"]["xAxes"][0]["ticks"]["suggestedMax"] = max_val * 1.35

    return cfg

    html = f"<table width='100%' cellpadding='0' cellspacing='0' border='0' style='border-collapse: collapse; font-size: 12px; font-family: Calibri, Arial, sans-serif; margin: 0 auto;'>"
    html += "<thead><tr>"
    for col in df.columns:
        html += f"<th style='background-color: {header_bg}; color: #fff; padding: 8px; border: 1px solid #ddd; text-align: center;'>{col}</th>"
    html += "</tr></thead><tbody>"

    for i, row in df.iterrows():
        bg = ModernColors.GRAY_LIGHT if zebra and i % 2 == 0 else ModernColors.WHITE
        html += f"<tr style='background-color: {bg};'>"
        for j, val in enumerate(row):
            v_str = str(val)
            color = ModernColors.ORANGE if "⚠️" in v_str else ModernColors.GRAY_DARK
            font_weight = "bold" if ("⚠️" in v_str or j == 0) else "normal"

            if "Aging" in df.columns and j == list(df.columns).index("Aging"):
                try:
                    if int(val) >= 30:
                        color = "#ffffff"
                        bg_cell = ModernColors.DANGER
                        font_weight = "bold"
                    else:
                        bg_cell = "transparent"
                except:
                    bg_cell = "transparent"
            else:
                bg_cell = ModernColors.WARNING_LIGHT if "⚠️" in v_str else "transparent"

            align = "left" if j == 0 else "center"
            html += f"<td style='padding: 8px; border: 1px solid #ddd; text-align: {align}; color: {color}; font-weight: {font_weight}; background-color: {bg_cell};'>{v_str}</td>"
        html += "</tr>"
    html += "</tbody>"

    if show_totals:
        html += f"<tfoot style='background-color: {ModernColors.GRAY_LIGHT}; font-weight: bold; color: {ModernColors.PRIMARY_DARK}; border-top: 2px solid {ModernColors.PRIMARY_DARK};'><tr>"
        for j, col_name in enumerate(df.columns):
            col_lower = str(col_name).strip().lower()
            if j == 0:
                val = "TOTAL"
                align = "left"
            elif "backlog acumulado" in col_lower:
                last_val = df.iloc[-1][col_name]
                val = int(last_val) if pd.notna(last_val) else 0
                align = "center"
            elif pd.api.types.is_numeric_dtype(df[col_name]):
                if "%" in col_lower or col_name == "%":
                    val = "100.0"
                else:
                    val = int(df[col_name].sum()) if sum(df[col_name]) == int(
                        sum(df[col_name])) else round(df[col_name].sum(), 1)
                align = "center"
            else:
                val = "—"
                align = "center"
            html += f"<td style='padding: 8px; border: 1px solid #ddd; text-align: {align};'>{val}</td>"
        html += "</tr></tfoot>"

    html += "</table>"
    return html

def create_raw_data_sheet(wb, df, sheet_name, q=None):
    log_q(q, f"   Criando aba: {sheet_name}...")
    ws = wb.create_sheet(sheet_name)
    if df.empty:
        ws.append(["Nenhum dado encontrado"])
        return
    df_export = df.copy()
    for col in df_export.select_dtypes(include=['datetime64[ns, UTC]', 'datetimetz']).columns:
        df_export[col] = df_export[col].dt.tz_localize(None)
    for col in df_export.columns:
        if df_export[col].apply(lambda x: isinstance(x, (list, dict))).any():
            df_export[col] = df_export[col].astype(str)
    df_export = df_export.fillna("")

    headers = list(df_export.columns)
    ws.append(headers)
    header_fill = PatternFill(start_color="2C3E50",
                              end_color="2C3E50", fill_type="solid")
    header_font = Font(color="FFFFFF", bold=True)
    for cell in ws[1]:
        cell.fill = header_fill
        cell.font = header_font

    for row in df_export.itertuples(index=False, name=None):
        ws.append(row)
    ws.auto_filter.ref = ws.dimensions
    ws.freeze_panes = "A2"


def create_executive_summary_sheet(wb, df_reqs, df_incs=None, sla_days_reqs=10, sla_days_incs=7):

def enviar_email_relatorio(html_content, caminho_anexo="", q=None):
    log_q(q, "\n--- Preparando envio de E-mail via Outlook ---")

    # Tratamento Rigoroso: Evita TypeError caso os.getenv retorne None
    safe_destinatario = str(destinatario) if destinatario else ""
    safe_copia = str(copia) if copia else ""
    safe_titulo = str(titulo) if titulo else "Relatório diário de tickets - Data: "
    
    try:
        safe_importance = int(importance) if importance else 2
    except (ValueError, TypeError):
        safe_importance = 2

    try:
        log_q(q, "Instanciando cliente de E-mail...")
        outlook = win32com.client.Dispatch("outlook.application")

        mail = outlook.CreateItem(0)
        
        if safe_destinatario:
            mail.To = safe_destinatario
        else:
            log_q(q, "⚠️ Aviso: Variável 'destinatario' está vazia. O envio pode falhar.")
            
        mail.Importance = safe_importance
        mail.Subject = safe_titulo + f"{datetime.now().strftime('%d/%m/%Y')}"
        mail.HTMLBody = html_content

        if safe_copia:
            mail.CC = safe_copia

        # Checagem de segurança dupla para o anexo
        if caminho_anexo and os.path.exists(caminho_anexo):
            absolute_path = os.path.abspath(caminho_anexo)
            mail.Attachments.Add(absolute_path)

        mail.Send()
        
        if caminho_anexo and os.path.exists(caminho_anexo):
            log_q(q, f"✅ E-mail enviado com sucesso (com anexo: {caminho_anexo})!")
        else:
            log_q(q, "✅ E-mail enviado com sucesso (sem anexo)!")

    except Exception as e:
        log_q(q, f"❌ Erro crítico ao enviar e-mail. Erro: {e}")

# MAIN SIMPLES COM AS VARIÁVEIS DINÂMICAS INJETADAS

def main_simples(configs, q=None):
    RUN_AT = datetime.now()

    log_q(q, "\n" + "=" * 80)
    log_q(q, "🚀 ELITE MASTER V4.1 - GERAÇÃO DE RELATÓRIO EXCEL + DASHBOARD HTML AVANÇADO (BLUE)")
    log_q(q, "=" * 80)

    try:
        _ensure_dir(OUTPUT_FILE)
        _ensure_dir(OUTPUT_HTML)

        # Extraindo SLAs das configs dinâmicas
        sla_reqs = int(configs.get("sla_days_reqs", 10))
        sla_incs = int(configs.get("sla_days_incs", 7))
        sla_ptask = int(configs.get("sla_days_ptask", 10))

        # CARREGAMENTO DINÂMICO
        df_reqs = load_prepare_reqs(SOURCE_FILE, "REQs", now=RUN_AT, q=q)
        df_incs = load_prepare_incs(SOURCE_INCS, "INCs", now=RUN_AT, q=q)

        # CARREGAMENTO: PTASK
        df_ptask = load_prepare_ptask(SOURCE_PTASK, sla_ptask, now=RUN_AT, q=q)

        df_reqs_clean = df_reqs.loc[:, ~df_reqs.columns.duplicated()].copy()
        df_incs_clean = df_incs.loc[:, ~df_incs.columns.duplicated()].copy()

        # ⚠️ CRIAÇÃO DA BASE BRUTA DINÂMICA UNIFICADA PARA ANEXO
        dynamic_date_str = RUN_AT.strftime("%d_%m_%Y")
        RAW_DATA_ATTACHMENT = os.path.join(os.path.dirname(
            OUTPUT_FILE) or ".", f"Base_Bruta_Consolidada_{dynamic_date_str}.xlsx")

        log_q(
            q, f"\n💾 Gerando anexo de dados brutos unificados: {RAW_DATA_ATTACHMENT}")
        with pd.ExcelWriter(RAW_DATA_ATTACHMENT, engine='openpyxl') as writer:
            if not df_reqs.empty:
                df_reqs.to_excel(writer, sheet_name='REQs', index=False)
            if not df_incs.empty:
                df_incs.to_excel(writer, sheet_name='INCs', index=False)
            if not df_ptask.empty:
                df_ptask.to_excel(writer, sheet_name='PTASK', index=False)

        # GERAÇÃO DO EXCEL (Se Escolhido no App)
        modo = configs.get("tipo_relatorio", "simples")
        if modo == 'completo':
            log_q(q, f"\n📝 Criando workbook Excel Completo...")
            wb = openpyxl.Workbook()
            wb.remove(wb.active)

            create_executive_summary_sheet(
                wb, df_reqs, df_incs=df_incs, sla_days_reqs=sla_reqs, sla_days_incs=sla_incs)
            create_temporal_analysis_sheet(wb, df_reqs, df_incs=df_incs, sla_days_reqs=sla_reqs,
                                           sla_days_incs=sla_incs, meses_extracao_override=None)
            create_subject_analysis_sheet(wb, df_reqs)

            if configs.get("enable_prb_abertos", True):
                create_prb_abertos_sheet(wb, df_reqs, df_incs)

            create_stagnation_sheet(wb, df_reqs=df_reqs, df_incs=df_incs,
                                    sla_days_reqs=sla_reqs, sla_days_incs=sla_incs)
            create_req_inc_unified_full_final_sheet(wb, df_reqs, df_incs)
            create_aging_sheet(wb, df_reqs, df_incs=df_incs, sla_days_reqs=sla_reqs,
                               sla_days_incs=sla_incs, top_n_detail=200)
            create_analyst_analysis_sheet(wb, df_reqs, df_incs=df_incs, top_n_analysts=20, sla_days_reqs=sla_reqs,
                                          sla_days_incs=sla_incs, months_limit=12, meses_extracao_override=None)

            create_raw_data_sheet(wb, df_reqs, "DADOS_BRUTOS_REQs", q=q)
            create_raw_data_sheet(wb, df_incs, "DADOS_BRUTOS_INCs", q=q)

            if not df_ptask.empty:
                create_raw_data_sheet(wb, df_ptask, "DADOS_BRUTOS_PTASK", q=q)

            log_q(q, f"\n💾 Salvando relatório Excel em: {OUTPUT_FILE}")
            wb.save(OUTPUT_FILE)
        else:
            log_q(
                q, "\n⏩ Pulando a geração do Excel completo (Modo Apenas HTML selecionado).")

        # HTML
        generate_advanced_html_report(
            df_reqs, df_incs, df_ptask, configs, q, OUTPUT_HTML)

        log_q(q, "\n" + "=" * 80)
        log_q(q, "✅ ROTINA CONCLUÍDA COM SUCESSO!")
        log_q(q, "=" * 80)
        log_q(q, f"\n📊 RESULTADOS:")

        if modo == 'completo':
            log_q(q, f"   📁 Relatório Excel Geral: {OUTPUT_FILE}")

        log_q(q, f"   📎 Arquivo Bruto Gerado (Anexo): {RAW_DATA_ATTACHMENT}")
        log_q(q, f"   📧 Dashboard HTML:  {OUTPUT_HTML}")
        log_q(
            q, f"   📈 Registros processados: REQs={len(df_reqs)}, INCs={len(df_incs)}, PTASK={len(df_ptask) if not df_ptask.empty else 0}")

        # ENVIO AUTOMÁTICO DE E-MAIL
        try:
            with open(OUTPUT_HTML, 'r', encoding='utf-8') as f:
                html_content_to_send = f.read()

            caminho_anexo_email = RAW_DATA_ATTACHMENT if configs.get(
                "attach_excel_to_email") else ""
            enviar_email_relatorio(html_content_to_send,
                                   caminho_anexo=caminho_anexo_email, q=q)

        except Exception as e:
            log_q(q, f"⚠️  Falha no processo de envio: {e}")

    except FileNotFoundError as e:
        log_q(q, f"\n❌ ERRO: {e}")
    except Exception as e:
        import traceback
        log_q(q, f"\n❌ ERRO INESPERADO: {e}")
        log_q(q, traceback.format_exc())
